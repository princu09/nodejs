Mongoose Connection to MongoDB 
const mongoose = require('mongoose');

// Connection Creation
mongoose.connect("mongodb://localhost:27017/nfg", {
    useNewUrlParser: true,
    useUnifiedTopology: true,
}).then(() => console.log("Connection Successfull...")).catch((err) => console.log(err));
Mongoose Create Schema 
// Create a Schema
const clientListSchema = new mongoose.Schema({
    name : {
        type : String,
        required : true
    },
    age : Number,
    city : String,
    date : {
        type : Date,
        default : Date.now,
    },
    author : String,
    active : Boolean,
})
Create Collection 
// Collection Create
const Client = new mongoose.model("PlayList" , clientListSchema);
Create Document or Insert
